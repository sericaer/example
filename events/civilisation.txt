country_event = {

	id = 3904

	trigger = {
		NOT = { capital_scope = { civilization_value = 50 } } 
		NOT = { has_country_modifier = "outside_influences"  }		
		NOT = { has_country_modifier = "stick_to_old_ways"  }
		OR = {
			government = religious_tribe
			government = military_tribe
			government = civic_tribe
			government = trade_tribe
		}
	}


	mean_time_to_happen = {
		months = 24
	}

	title = "EVTNAME3904"
	desc = "EVTDESC3904"

	option = {
		name = "EVTOPTA3904"
		capital_scope = { civilization_value  = 1 }
		ruler = {
			popularity =  -30
		}
		any_character = {
			limit = { has_job = yes }
			loyalty = -10
		}
		
		add_country_modifier = {
			name = "outside_influences"
			duration = 730
		}		
	}

	option = {
		name = "EVTOPTB3904"
		ruler = {
			popularity =  10
		}
		any_character = {
			limit = { has_job = yes }
			loyalty = 10
		}		
		add_country_modifier = {
			name = "stick_to_old_ways"
			duration = 730
		}		
	}
}
